# --gpus
#   default: none, required
#   values: one or more GPU ids, e.g.:
#     0
#     0 1
#     0 1 2 3
#   note: one Optuna worker is launched per listed GPU
#
# --n_trials
#   default: none, required
#   values: positive int
#   note: total trial budget across all GPU workers
#
# --storage
#   default: none, required
#   values: any shared Optuna storage URL, e.g.:
#     sqlite:////absolute/path/to/kcatnet_optuna.db
#   note: all workers must point to the same study storage
#
# --stagger_seconds
#   default: 3.0
#   values: non-negative float
#   note: delay between worker launches to reduce storage startup contention
#
# All other flags mirror tune_optuna.py.
# Practical note:
#   each worker gets CUDA_VISIBLE_DEVICES set to a single GPU and runs tune_optuna.py with --device cuda:0
#   this keeps each trial single-GPU while multiple trials run in parallel

# --study_name / --storage
#   required unless you provide --hparams_json
#   behavior: script loads the best trial from this Optuna study/storage and resolves train hparams
#   note: if you accidentally pass a name like <study>_best_hparams, the launcher now auto-falls back
#         to <study> when possible
#
# --live_train_logs / --no-live_train_logs
#   default: live_train_logs enabled
#   behavior: stream train subprocess output to terminal (shows tqdm epoch/batch bars)
#
# --live_predict_logs / --no-live_predict_logs
#   default: live_predict_logs enabled
#   behavior: stream prediction subprocess output to terminal
#
# --hparams_json
#   optional alternative to storage extraction
#   values: path to a JSON file with tuned hparams
#
# --gpus
#   required
#   values: one or more GPU ids, e.g.:
#     0
#     0 1
#     0 1 2 3
#   note: one retrain job is assigned per GPU at a time using CUDA_VISIBLE_DEVICES
#
# --split_groups / --threshold / --thresholds / --seeds
#   values: same split selectors as run_split_benchmarks.py, with one or more seeds
#
# --output_root
#   default: /home/adhil/github/EMULaToR/data/processed/baselines/KcatNet/retrain_from_optuna/<study_name>
#   values: custom output root if you want results elsewhere
#   note: on rerun (without --overwrite), existing bestmodel.pth is reused and only missing predict/metrics steps are executed
#
# Per-run outputs (each split/seed has its own directory):
#   train/
#     bestmodel.pth
#     checkpoint_last.pt
#   predictions/
#     train_predictions.csv
#     val_predictions.csv
#     test_predictions.csv
#     *_predictions_metrics.csv
#   metrics/
#     tvt_metrics_long.csv     (columns include r2/pcc/scc/mse/rmse/mae)
#     tvt_metrics_wide.csv
#   logs/
#     train.log
#     predict_train.log
#     predict_val.log
#     predict_test.log
#
# Global outputs at output_root:
#   selected_hparams.json
#   planned_runs.csv
#   runs_status.csv
#   all_tvt_metrics.csv
#   aggregate_tvt_metrics.csv
#   aggregate_test_metrics_ranked.csv
